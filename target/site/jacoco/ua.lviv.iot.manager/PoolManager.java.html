package ua.lviv.iot.manager;

import ua.lviv.iot.models.Pool;

import java.util.ArrayList;
import java.util.List;

public class PoolManager {

    public List selectionSortByDecreaseOfCapacity(ArrayList<Double> listOfPoolsCapacity) {



        for (int i = 0; i < listOfPoolsCapacity.size(); i++) {
            double max = listOfPoolsCapacity.get(i);
            int maxIndex = i;
            for (int j = i + 1; j < listOfPoolsCapacity.size(); j++) {

                if (max < listOfPoolsCapacity.get(j)) {
                    max = listOfPoolsCapacity.get(j);
                    maxIndex = j;
                }
            }

            double tempo = listOfPoolsCapacity.get(i);
            listOfPoolsCapacity.set(i, listOfPoolsCapacity.get(maxIndex));
            listOfPoolsCapacity.set(maxIndex, tempo);

        }
        return listOfPoolsCapacity;
    }

    public static void main(String[] args) {
        PoolManager poolManager = new PoolManager();

        ArrayList<Pool> listOfPools = new ArrayList();

        ArrayList listOfPoolsCapacity = new ArrayList();

        listOfPools.add(new Pool("Bandery 21", 400, 50));
        listOfPools.add(new Pool("Horodotska 11", 2000, 400));
        listOfPools.add(new Pool("Varshavska 321", 1400, 340));
        listOfPools.add(new Pool("Mazepy 1", 50, 10));
        listOfPools.add(new Pool("Polya 22", 100, 20));

        for (int i = 0; i < listOfPools.size(); i++) {
            listOfPoolsCapacity.add(listOfPools.get(i).getCapacity());
        }

        List sortedListOfPoolsCapacity = poolManager.selectionSortByDecreaseOfCapacity(listOfPoolsCapacity);

        System.out.println(sortedListOfPoolsCapacity);


    }
}
